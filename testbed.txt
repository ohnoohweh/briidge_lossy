# Test cases

# Test pattern:
#
# 1. Instanciate a bounce back server, overlay_tty.py. It bounces back received payload with a modified content to clearly trace it along the path
#    on incomming payload 01 30 it responds with outgoing payload 02 30
#
# 2. Instanciate python ObstacleBridge.py as an outgoing connection server
#
# 3. Instanciate python ObstacleBridge.py as an incomming connection server
#
# wait for 2 service to allow udp_bidirectional_main to connect
#
# 4. Instanciate a outgoing connection server to ObstacleBridge.py as an incomming connection server
#    Send payload as 01 30
#    Within 1s expect incomming payload 02 30
#
# 5. Kill ObstacleBridge.py as an outgoing connection server
#
# 6. Expect within 30s switch to disconnected
#
# 7. Instanciate python ObstacleBridge.py as an outgoing connection server
#
# wait for 2 service to allow udp_bidirectional_main to connect
# 
# 8. Send payload as 01 31
#    Within 1s expect incomming payload 02 31
# 
# 9. Kill ObstacleBridge.py as an incomming connection server
# 
# 10. Expect within 30s switch to disconnected
#
# 11. Instanciate python ObstacleBridge.py as an incomming connection server
#
# wait for 2 service to allow udp_bidirectional_main to connect
#
# 12. Send payload as 01 32
#    Within 1s expect incomming payload 02 32
#
# Disconnect can be tested using --admin-web 
# for python ObstacleBridge.py as an outgoing connection server -admin-web-port 18080
# for python ObstacleBridge.py as an outgoing connection server -admin-web-port 18090

Certificate for Server with static IPv4 and IPv6
san.cnf: Replace <IPv?> with static addresses
[req]
prompt = no
distinguished_name = dn
x509_extensions = ext

[dn]
CN = <IPv4>

[ext]
subjectAltName = @alt_names

[alt_names]
IP.1 = <IPv4>
IP.2 = <IPv6>

openssl req -x509 -newkey rsa:2048 -keyout key.pem -out cert.pem -days 365 -nodes -config san.cnf
